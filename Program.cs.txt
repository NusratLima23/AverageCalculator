using System;

class Program
{
    static void Main()
    {
        int total = 0;

        for (int i = 1; i <= 5; i++)
        {
            Console.Write("Enter grade " + i + ": ");
            int grade = Convert.ToInt32(Console.ReadLine());
            total += grade;
        }

        double average = total / 5.0;
        Console.WriteLine("The average grade is: " + average);
    }
}
